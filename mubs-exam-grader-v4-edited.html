WebSocket Connection

ws://localhost:8765 — Not connected (simulated mode)

In production, run mubs_agent.py inside the student's VM. The grader connects to it via WebSocket and switches from "Simulated" to "Live" mode — awarding marks based on actual VM state rather than typed commands.

State Payload Format

// JSON sent by VM agent every 2 seconds { "timestamp": "2026-03-26T08:14:05Z", "student_id": "22/U/04567", "vm_id": "LAB-PC-07", "states": { "apache_installed": true, "apache_running": true, "apache_enabled": false, "index_html_exists": true, "index_html_content": "<html>...</html>", "localhost_responds": true, "http_status": 200 } }

Deploy this Python script inside the student's Ubuntu VM. It polls the system state every 2 seconds and posts to the grader over HTTP or WebSocket.

#!/usr/bin/env python3 # mubs_agent.py — MUBS VM State Reporter # Drop into student VM: python3 mubs_agent.py import subprocess, json, time, urllib.request, os GRADER_URL = "http://HOST_IP:8080/state" STUDENT_ID = os.environ.get("STUDENT_ID", "unknown") VM_ID = os.environ.get("VM_ID", "unknown") INTERVAL = 2 # seconds between polls def run(cmd): try: r = subprocess.run(cmd, shell=True, capture_output=True, text=True, timeout=5) return r.stdout.strip(), r.returncode except: return "", 1 def collect_state(): installed, _ = run("dpkg -l apache2 2>/dev/null | grep '^ii'") running, rc1 = run("systemctl is-active apache2") enabled, rc2 = run("systemctl is-enabled apache2") idx_html, _ = run("cat /var/www/html/index.html 2>/dev/null") curl_out, crc = run("curl -s -o /dev/null -w '%{http_code}' localhost") return { "apache_installed": bool(installed), "apache_running": running == "active", "apache_enabled": running == "enabled", "index_html_exists": bool(idx_html), "index_html_content": idx_html[:4000], "localhost_responds": curl_out == "200", "http_status": int(curl_out) if curl_out.isdigit() else 0 } def post_state(payload): data = json.dumps(payload).encode() req = urllib.request.Request(GRADER_URL, data=data, headers={"Content-Type": "application/json"}) try: urllib.request.urlopen(req, timeout=3) except: pass # silent — don't interrupt student print(f"MUBS Agent running. Reporting to {GRADER_URL}") while True: state = {"timestamp": time.strftime("%Y-%m-%dT%H:%M:%SZ"), "student_id": STUDENT_ID, "vm_id": VM_ID, "states": collect_state()} post_state(state) time.sleep(INTERVAL)
